# L7-3: Review Rule Customization

> Build review standards that match team specifications

## Section Overview

Every team has its own code standards and quality requirements. How can you make AI code review meet your team's specific needs? This lesson will teach you how to customize review rules and integrate team standards.

By the end of this lesson, you will learn:

## 1. Rule System Design

### 1.1 Rule Hierarchy

```
Rule Hierarchy Structure:

Level 1: Base Rules (Must Follow)
├── Security Rules (Critical)
├── Syntax Rules (Error)
└── Type Rules (Error)

Level 2: Team Standards (Should Follow)
├── Naming Conventions (Warning)
├── Code Style (Warning)
└── Documentation Standards (Warning)

Level 3: Best Practices (Recommended)
├── Design Patterns (Suggestion)
├── Performance Optimization (Suggestion)
└── Readability (Suggestion)

Level 4: Personalized Rules (Optional)
├── Team Preferences (Info)
└── Project-Specific (Info)
```

### 1.2 Rule Configuration Format


## 2. Rule Priority Management

### 2.1 Priority Strategy

```typescript
interface RulePriority {
  // Rule ID
  ruleId: string;
  
  // Priority level
  level: 'critical' | 'high' | 'medium' | 'low' | 'info';
  
  // Whether blocking (prevents merge)
  blocking: boolean;
  
  // Auto-fix
  autoFix: boolean;
  
  // Ignore conditions
  ignorePatterns?: string[];
  
  // Custom message
  customMessage?: string;
}

// Priority configuration
const priorityConfig: RulePriority[] = [
  {
    ruleId: 'sql-injection',

    blocking: true,

    ruleId: 'xss-risk',
    level: 'critical',
    blocking: true,
    autoFix: false
  },
  {
    ruleId: 'n-plus-one',
    level: 'warning',
    blocking: false,
    autoFix: false
  },
  {
    ruleId: 'naming-convention',
    level: 'warning',
    blocking: false,
    autoFix: true
  }

```

### 2.2 Rule Override Strategy

```
Rule Override Scenarios:

Scenario 1: Global Rules + Project-Specific Rules
├── .ai-review/rules.yaml (global)
└── project/.ai-review/rules.yaml (project-specific)
    Project rules override global rules

Scenario 2: Inheritance + Extension
base-rules.yaml
  └── extends: base-rules
      rules:
        - disable: rule-a
        - add: rule-b

Scenario 3: Environment Differences
development: Enable all rules
production: Enable only critical and high
```

### 3.2 Standards Sync Mechanism

```typescript
class TeamStandardSync {
  // Sync rules from ESLint configuration
  async syncFromESLint(configPath: string): Promise<Rule[]> {
    const eslintConfig = await this.loadConfig(configPath);
    
    return eslintConfig.rules.map(([ruleId, severity]) => ({
      id: ruleId,
      level: this.mapSeverity(severity),
      source: 'eslint',
      enabled: severity !== 'off'
    }));
  }
  
  // Extract rules from coding standards document
  async syncFromStyleGuide(docPath: string): Promise<Rule[]> {
    const doc = await this.loadDocument(docPath);
    
    // Use AI to extract rules
    const prompt = `
Extract auto-checkable rules from the following coding standards document:

${doc}

Output format:
- Rule name
- Rule description
- Check method
- Severity
    `;
    
    const rules = await this.aiExtractRules(prompt);
    return rules;
  }
  
  // Learn from historical review records
  async learnFromHistory(reviews: CodeReview[]): Promise<Rule[]> {
    // Analyze issue patterns
    const patterns = this.analyzePatterns(reviews);
    
    // Generate new rules
    return patterns.map(pattern => ({
      id: `learned-${pattern.id}`,
      name: pattern.name,
      level: pattern.severity,
      pattern: pattern.regex,
      message: pattern.suggestion,
      source: 'learned'
    }));
  }
}
```

```yaml
# Basic Settings
settings:
  # Review languages
  languages:
    - typescript
    - javascript
    - react
  
  # Exclude paths
  exclude:
    - 'node_modules/**'
    - 'dist/**'
    - '**/*.test.ts'
    - '**/*.spec.ts'
  
  # Include paths (takes priority over exclude)
  include:
    - 'src/**'
  
  # Max file size (KB)
  max_file_size: 500
  
  # Parallel review count
  max_concurrent: 3

rules:
  # Extend preset rule sets
  extends:
    - '@ai-review/security'
    - '@ai-review/performance'
    - '@ai-review/react'
  
  # Disable certain rules
  disable:
    - 'prefer-default-export'
    - 'no-console'
  
  # Custom rules
  custom:
    # Team-specific naming conventions
    - id: team-naming
      name: Team Naming Conventions

      pattern: |
        Functions use camelCase
        Classes use PascalCase
        Constants use UPPER_SNAKE_CASE
        Private methods use _ prefix

    # Specific business rules
    - id: no-direct-api-call
      name: No Direct API Calls
```

### 4.2 Rule Validation

```typescript
class RuleValidator {
  // Validate rule configuration
  validate(config: ReviewConfig): ValidationResult {
    const errors: string[] = [];
    const warnings: string[] = [];
    
    // Check rule ID uniqueness
    const ruleIds = new Set<string>();
    for (const rule of config.rules.custom) {
      if (ruleIds.has(rule.id)) {
        errors.push(`Duplicate rule ID: ${rule.id}`);
      }
      ruleIds.add(rule.id);
    }
    
    // Check rule completeness
    for (const rule of config.rules.custom) {
      if (!rule.name) {
        errors.push(`Rule ${rule.id} missing name`);
      }
      if (!rule.severity) {
        errors.push(`Rule ${rule.id} missing severity level`);
      }
      if (!rule.pattern && !rule.check) {
        warnings.push(`Rule ${rule.id} missing check logic`);
      }
    }
    
    // Check workflow configuration
    if (config.workflow.pre_commit.enabled && 
        config.workflow.pre_commit.rules.length === 0) {
      warnings.push('pre_commit enabled but no rules configured');
    }
    
    return {
      valid: errors.length === 0,
      errors,
      warnings
    };
  }
  
  // Test rules
  async testRule(rule: Rule, testCases: TestCase[]): Promise<TestResult> {
    const results: TestCaseResult[] = [];
    
    for (const testCase of testCases) {
      const violations = await this.applyRule(rule, testCase.code);
      const expected = testCase.shouldTrigger;
      const actual = violations.length > 0;
      
      results.push({
        name: testCase.name,
        passed: actual === expected,
        expected,
        actual,
        violations
      });
    }
    
    return {
      ruleId: rule.id,
      total: testCases.length,
      passed: results.filter(r => r.passed).length,
      failed: results.filter(r => !r.passed).length,
      details: results
    };
  }
}
```
