## 2. Version Control

### 2.1 Version Strategy

**Semantic Versioning**: MAJOR.MINOR.PATCH

- **MAJOR**: Incompatible API changes
- **MINOR**: Backward-compatible feature additions
- **PATCH**: Backward-compatible bug fixes

**Examples**:
- v1.0.0 - Initial release
- v1.1.0 - Added Vue support
- v1.1.1 - Fixed React 18 detection issue
- v2.0.0 - Rebuilt API, incompatible with v1.x

### 2.2 Version Management

```typescript
interface SkillVersion {
  version: string;
  changelog: string;
  breaking: boolean;
  deprecated?: boolean;
  migration?: string;
}

class SkillVersionManager {
  private versions: Map<string, SkillVersion[]> = new Map();
  
  // Publish new version
  publishVersion(skillId: string, version: SkillVersion): void {
    const skillVersions = this.versions.get(skillId) || [];
    
    // Check if version number is valid
    if (!this.isValidVersion(version.version)) {
      throw new Error(`Invalid version: ${version.version}`);
    }
    
    // Check if already exists
    if (skillVersions.some(v => v.version === version.version)) {
      throw new Error(`Version ${version.version} already exists`);
    }
    
    skillVersions.push(version);
    this.versions.set(skillId, skillVersions);
  }
  
  // Get latest version
  getLatestVersion(skillId: string): string | null {
    const versions = this.versions.get(skillId);
    if (!versions || versions.length === 0) return null;
    
    return versions
      .filter(v => !v.deprecated)
      .sort((a, b) => this.compareVersion(b.version, a.version))[0]?.version || null;
  }
  
  // Version comparison
  private compareVersion(v1: string, v2: string): number {
    const parts1 = v1.split('.').map(Number);
    const parts2 = v2.split('.').map(Number);
    
    for (let i = 0; i < Math.max(parts1.length, parts2.length); i++) {
      const p1 = parts1[i] || 0;
      const p2 = parts2[i] || 0;
      
      if (p1 > p2) return 1;
      if (p1 < p2) return -1;
    }
    
    return 0;
  }
}
```

## 4. Practice: Establish Team Skill Library

### 4.1 Initialize Repository

```bash
# Create Skill library
mkdir team-skill-library
cd team-skill-library
git init

# Create directory structure
mkdir -p categories/{code-review,code-generation,documentation}
mkdir -p shared/{templates,knowledge,utils}
mkdir -p registry

# Create configuration file
cat > package.json << 'EOF'
{
  "name": "@company/skill-library",
  "version": "1.0.0",
  "description": "Team Skill Library",
  "scripts": {
    "validate": "skill-cli validate",
    "publish": "skill-cli publish",
    "test": "skill-cli test"
  }
}
EOF
```

### 4.2 Add Skill

```bash
# Add React Reviewer Skill
cd categories/code-review
mkdir react-reviewer
cd react-reviewer

# Create Skill file
cat > skill.yaml << 'EOF'
skill:
  id: react-reviewer
  name: React Code Review Expert
  version: 1.0.0
  category: code-review
  language: [typescript, javascript]
  
  description: |
    Professional React code review Skill covering:
    - React best practices
    - Hooks usage standards
    - Performance optimization suggestions
    - TypeScript type checking
  
  author: Frontend Team
  tags: [react, review, frontend]
EOF

# Create Prompt file
cat > prompt.md << 'EOF'

## Role
You are a React development expert with 5+ years of experience...

## Review Dimensions
...
EOF
```
