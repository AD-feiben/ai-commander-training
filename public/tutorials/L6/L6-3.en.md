## 2. Communication Protocol Design

### 2.1 Communication Patterns

```
Pattern 1: Point-to-Point
┌─────────┐         ┌─────────┐
│ Agent A │←───────→│ Agent B │
└─────────┘         └─────────┘
Use case: One-to-one communication, e.g., task handoff

Pattern 2: Publish-Subscribe (Pub-Sub)
┌─────────┐
│ Agent A │──Publish──→ Topic X
└─────────┘      ↓
            ┌─────────┐
            │ Message │
            │   Bus   │
            └────┬────┘
                 ↓
     ┌───────────┼───────────┐
     ↓           ↓           ↓
┌─────────┐ ┌─────────┐ ┌─────────┐
│ Agent B │ │ Agent C │ │ Agent D │
└─────────┘ └─────────┘ └─────────┘
Use case: One-to-many broadcast, e.g., state updates

Pattern 3: Request-Response
┌─────────┐    Request    ┌─────────┐
│ Agent A │───────────────→│ Agent B │
└─────────┘                └────┬────┘
     ↑                      │
     └────────── Response ───┘
Use case: Synchronous calls, e.g., get information

Pattern 4: Event-Driven
┌─────────┐    Event     ┌─────────┐
│ Agent A │──────────────→│  Event  │
└─────────┘                │   Bus   │
                           └────┬────┘
              ┌─────────────┼─────────────┐
              ↓             ↓             ↓
         ┌─────────┐  ┌─────────┐  ┌─────────┐
         │ Handler │  │ Handler │  │ Handler │
         └─────────┘  └─────────┘  └─────────┘
Use case: Asynchronous processing, e.g., task completion notification
```

### 2.3 Message Bus Implementation

```typescript
class MessageBus {
  private subscribers: Map<string, Set<(message: Message) => void>> = new Map();
  private messageQueue: Message[] = [];
  private messageHistory: Message[] = [];
  private maxHistorySize = 1000;

  // Subscribe to messages
  subscribe(
    messageType: string,
    handler: (message: Message) => void,
    filter?: (message: Message) => boolean
  ): () => void {
    const wrappedHandler = (message: Message) => {
      if (!filter || filter(message)) {
        handler(message);
      }
    };

    if (!this.subscribers.has(messageType)) {
      this.subscribers.set(messageType, new Set());
    }

    this.subscribers.get(messageType)!.add(wrappedHandler);

    // Return unsubscribe function
    return () => {
      this.subscribers.get(messageType)?.delete(wrappedHandler);
    };
  }

  // Publish message
  publish(message: Message): void {
    // Add to queue
    this.messageQueue.push(message);

    // Add to history
    this.messageHistory.push(message);
    if (this.messageHistory.length > this.maxHistorySize) {
      this.messageHistory.shift();
    }

    // Dispatch immediately
    this.dispatch(message);
  }

  // Dispatch message
  private dispatch(message: Message): void {
    const handlers = this.subscribers.get(message.type);
    if (handlers) {
      handlers.forEach(handler => {
        try {
          handler(message);
        } catch (error) {
          console.error(`Message handling failed: ${message.id}`, error);
        }
      });
    }

    // Broadcast messages also trigger wildcard subscriptions
    const wildcardHandlers = this.subscribers.get('*');
    if (wildcardHandlers) {
      wildcardHandlers.forEach(handler => {
        try {
          handler(message);
        } catch (error) {
          console.error(`Message handling failed: ${message.id}`, error);
        }
      });
    }
  }

  // Send message and wait for response
  async sendAndWait(
    message: Message,
    timeout: number = 30000
  ): Promise<Message> {
    return new Promise((resolve, reject) => {
      const correlationId = message.id;
      let timeoutId: NodeJS.Timeout;

      const unsubscribe = this.subscribe(
        'query.response',
        (response: Message) => {
          if (response.correlationId === correlationId) {
            clearTimeout(timeoutId);
            unsubscribe();
            resolve(response);
          }
        }
      );

      timeoutId = setTimeout(() => {
        unsubscribe();
        reject(new Error(`Request timeout: ${correlationId}`));
      }, timeout);

      this.publish(message);
    });
  }

  // Get message history
  getHistory(filter?: (message: Message) => boolean): Message[] {
    if (filter) {
      return this.messageHistory.filter(filter);
    }
    return [...this.messageHistory];
  }

  // Clear history
  clearHistory(): void {
    this.messageHistory = [];
  }
}
```

### 3.2 State Version Control

```typescript
interface StateVersion {
  version: number;
  timestamp: number;
  agent: string;
  changes: StateChange[];
}

interface StateChange {
  path: string;
  operation: 'set' | 'delete' | 'merge';
  oldValue?: any;
  newValue?: any;
}

class SharedState {
  private state: any = {};
  private versions: StateVersion[] = [];
  private currentVersion = 0;
  private messageBus: MessageBus;

  constructor(messageBus: MessageBus) {
    this.messageBus = messageBus;

    // Subscribe to state update messages
    this.messageBus.subscribe('state.updated', (message) => {
      this.handleRemoteUpdate(message.payload);
    });
  }

  // Get state
  get(path?: string): any {
    if (!path) return this.state;
    return this.getValueByPath(this.state, path);
  }

  // Set state (local modification)
  set(path: string, value: any, agent: string): void {
    const oldValue = this.getValueByPath(this.state, path);

    // Update state
    this.setValueByPath(this.state, path, value);

    // Create new version
    this.currentVersion++;
    const version: StateVersion = {
      version: this.currentVersion,
      timestamp: Date.now(),
      agent,
      changes: [{
        path,
        operation: 'set',
        oldValue,
        newValue: value
      }]
    };

    this.versions.push(version);

    // Broadcast update
    this.messageBus.publish({
      id: generateId(),
      type: 'state.updated',
      sender: agent,
      timestamp: Date.now(),
      payload: {
        entity: 'shared_state',
        property: path,
        oldValue,
        newValue: value,
        version: this.currentVersion
      },
      priority: 'high'
    });
  }

  // Handle remote update
  private handleRemoteUpdate(payload: any): void {
    const { property, newValue, version } = payload;

    // Check version conflict
    if (version <= this.currentVersion) {
      console.warn(`State version conflict: local ${this.currentVersion}, remote ${version}`);
      // Conflict resolution strategy...
      return;
    }

    // Apply update
    this.setValueByPath(this.state, property, newValue);
    this.currentVersion = version;
  }

  // Get version history
  getHistory(since?: number): StateVersion[] {
    if (since) {
      return this.versions.filter(v => v.timestamp >= since);
    }
    return [...this.versions];
  }

  // Rollback to specified version
  rollback(targetVersion: number, agent: string): void {
    const target = this.versions.find(v => v.version === targetVersion);
    if (!target) {
      throw new Error(`Version ${targetVersion} does not exist`);
    }

    // Rollback all changes
    for (const change of target.changes) {
      this.setValueByPath(this.state, change.path, change.oldValue);
    }

    // Broadcast rollback
    this.messageBus.publish({
      id: generateId(),
      type: 'state.rollback',
      sender: agent,
      timestamp: Date.now(),
      payload: {
        fromVersion: this.currentVersion,
        toVersion: targetVersion
      },
      priority: 'critical'
    });

    this.currentVersion = targetVersion;
  }

  // Helper method: Get value by path
  private getValueByPath(obj: any, path: string): any {
    const parts = path.split('.');
    let current = obj;

    for (const part of parts) {
      if (current === null || current === undefined) {
        return undefined;
      }
      current = current[part];
    }

    return current;
  }

  // Helper method: Set value by path
  private setValueByPath(obj: any, path: string, value: any): void {
    const parts = path.split('.');
    let current = obj;

    for (let i = 0; i < parts.length - 1; i++) {
      const part = parts[i];
      if (!(part in current)) {
        current[part] = {};
      }
      current = current[part];
    }

    current[parts[parts.length - 1]] = value;
  }
}
```

## 4. Conflict Resolution Strategies

### 4.1 Conflict Types

```
Conflict Types:

1. Concurrent Modification Conflict
   Agent A ──Read X=v1──→ Modify X=v2
   Agent B ──Read X=v1──→ Modify X=v3
   Result: X's value is uncertain

2. Dependency Violation Conflict
   Agent A ──Complete Task A (depended on by B)
   Agent B ──Start Task B (depends on A)
   Problem: If A fails, B's work is wasted

3. Resource Contention Conflict
   Agent A ──Modify File X
   Agent B ──Simultaneously modify File X
   Result: Modifications lost or conflict
```

### 4.2 Conflict Detection

```typescript
class ConflictDetector {
  private pendingChanges: Map<string, PendingChange> = new Map();

  // Detect conflict
  detectConflict(
    agentId: string,
    change: StateChange,
    currentState: any
  ): Conflict | null {
    const changeKey = `${change.path}`;

    // Check if other Agents are modifying the same location
    for (const [otherAgentId, pendingChange] of this.pendingChanges) {
      if (otherAgentId !== agentId) {
        for (const otherChange of pendingChange.changes) {
          if (this.isConflicting(change, otherChange)) {
            return {
              type: 'concurrent_modification',
              involvedAgents: [agentId, otherAgentId],
              details: {
                path: change.path,
                change1: change,
                change2: otherChange
              }
            };
          }
        }
      }
    }

    return null;
  }

  // Check if two changes conflict
  private isConflicting(change1: StateChange, change2: StateChange): boolean {
    // Same path or one contains the other
    const path1 = change1.path;
    const path2 = change2.path;

    return path1 === path2 ||
           path1.startsWith(path2 + '.') ||
           path2.startsWith(path1 + '.');
  }

  // Record pending changes
  recordPendingChange(agentId: string, changes: StateChange[]): void {
    this.pendingChanges.set(agentId, {
      agentId,
      changes,
      timestamp: Date.now()
    });
  }

  // Clear completed changes
  clearPendingChange(agentId: string): void {
    this.pendingChanges.delete(agentId);
  }
}

interface Conflict {
  type: 'concurrent_modification' | 'dependency_violation' | 'resource_contention';
  involvedAgents: string[];
  details: any;
}
```

### 4.3 Conflict Resolution Strategies

```typescript
class ConflictResolver {
  private messageBus: MessageBus;

  constructor(messageBus: MessageBus) {
    this.messageBus = messageBus;
  }

  // Resolve conflict
  async resolveConflict(conflict: Conflict): Promise<Resolution> {
    switch (conflict.type) {
      case 'concurrent_modification':
        return await this.resolveConcurrentModification(conflict);
      case 'dependency_violation':
        return await this.resolveDependencyViolation(conflict);
      case 'resource_contention':
        return await this.resolveResourceContention(conflict);
      default:
        throw new Error(`Unknown conflict type: ${conflict.type}`);
    }
  }

  // Strategy 1: Last Write Wins
  private async resolveConcurrentModification(
    conflict: Conflict
  ): Promise<Resolution> {
    const { change1, change2 } = conflict.details;

    // Choose the change with later timestamp
    const winner = change1.timestamp > change2.timestamp ? change1 : change2;
    const loser = change1.timestamp > change2.timestamp ? change2 : change1;

    return {
      strategy: 'last_write_wins',
      acceptedChange: winner,
      rejectedChange: loser,
      reason: `Selected change with later timestamp (${winner.timestamp} > ${loser.timestamp})`
    };
  }

  // Strategy 2: Negotiation
  private async resolveByNegotiation(
    conflict: Conflict
  ): Promise<Resolution> {
    const { involvedAgents, details } = conflict;

    // Send negotiation requests
    const proposals = await Promise.all(
      involvedAgents.map(agentId =>
        this.requestProposal(agentId, conflict)
      )
    );

    // Vote to select best proposal
    const votes = await this.collectVotes(involvedAgents, proposals);
    const winner = this.selectWinner(votes);

    return {
      strategy: 'negotiation',
      acceptedChange: winner.proposal,
      rejectedChanges: proposals.filter(p => p !== winner.proposal),
      reason: `Selected by voting, votes: ${winner.votes}`
    };
  }

  // Strategy 3: Coordinator Arbitration
  private async resolveByCoordinator(
    conflict: Conflict,
    coordinatorId: string
  ): Promise<Resolution> {
    // Send to coordinator for arbitration
    const decision = await this.messageBus.sendAndWait({
      id: generateId(),
      type: 'conflict.arbitration',
      sender: 'conflict_resolver',
      receiver: coordinatorId,
      timestamp: Date.now(),
      payload: conflict,
      priority: 'critical'
    });

    return decision.payload as Resolution;
  }

  private async requestProposal(agentId: string, conflict: Conflict): Promise<any> {
    const response = await this.messageBus.sendAndWait({
      id: generateId(),
      type: 'conflict.proposal_request',
      sender: 'conflict_resolver',
      receiver: agentId,
      timestamp: Date.now(),
      payload: conflict,
      priority: 'high'
    });

    return response.payload;
  }

  private async collectVotes(agents: string[], proposals: any[]): Promise<Vote[]> {
    const votes: Vote[] = [];

    for (const agentId of agents) {
      const vote = await this.messageBus.sendAndWait({
        id: generateId(),
        type: 'conflict.vote_request',
        sender: 'conflict_resolver',
        receiver: agentId,
        timestamp: Date.now(),
        payload: { proposals },
        priority: 'high'
      });

      votes.push({
        agentId,
        choice: vote.payload.choice
      });
    }

    return votes;
  }

  private selectWinner(votes: Vote[]): { proposal: any; votes: number } {
    const counts = new Map<any, number>();

    for (const vote of votes) {
      counts.set(vote.choice, (counts.get(vote.choice) || 0) + 1);
    }

    let winner: any;
    let maxVotes = 0;

    for (const [proposal, count] of counts) {
      if (count > maxVotes) {
        winner = proposal;
        maxVotes = count;
      }
    }

    return { proposal: winner, votes: maxVotes };
  }
}
```

### 5.2 Simple Majority Voting

```typescript
class MajorityVoting {
  private messageBus: MessageBus;

  async reachConsensus(
    topic: string,
    options: string[],
    participants: string[]
  ): Promise<ConsensusResult> {
    // Phase 1: Collect proposals
    const proposals = await this.collectProposals(participants, topic);

    // Phase 2: Vote
    const votes = await this.collectVotes(participants, proposals);

    // Phase 3: Count votes
    const result = this.countVotes(votes);

    // Check if majority threshold is reached
    const majorityThreshold = Math.ceil(participants.length / 2);

    if (result.winner.votes >= majorityThreshold) {
      return {
        topic,
        decision: result.winner.option,
        support: result.winner.votes / participants.length,
        votes: result.voteDistribution,
        consensus: true
      };
    } else {
      return {
        topic,
        decision: null,
        support: result.winner.votes / participants.length,
        votes: result.voteDistribution,
        consensus: false,
        reason: 'Majority threshold not reached'
      };
    }
  }

  private async collectProposals(
    participants: string[],
    topic: string
  ): Promise<string[]> {
    const proposals: string[] = [];

    for (const agentId of participants) {
      const response = await this.messageBus.sendAndWait({
        id: generateId(),
        type: 'consensus.propose',
        sender: 'coordinator',
        receiver: agentId,
        timestamp: Date.now(),
        payload: { topic },
        priority: 'high'
      });

      proposals.push(response.payload.proposal);
    }

    return [...new Set(proposals)]; // Deduplicate
  }

  private async collectVotes(
    participants: string[],
    proposals: string[]
  ): Promise<Vote[]> {
    const votes: Vote[] = [];

    for (const agentId of participants) {
      const response = await this.messageBus.sendAndWait({
        id: generateId(),
        type: 'consensus.vote',
        sender: 'coordinator',
        receiver: agentId,
        timestamp: Date.now(),
        payload: { proposals },
        priority: 'high'
      });

      votes.push({
        agentId,
        choice: response.payload.choice,
        reasoning: response.payload.reasoning
      });
    }

    return votes;
  }

  private countVotes(votes: Vote[]): VoteResult {
    const distribution = new Map<string, number>();

    for (const vote of votes) {
      distribution.set(
        vote.choice,
        (distribution.get(vote.choice) || 0) + 1
      );
    }

    let winner: { option: string; votes: number } = { option: '', votes: 0 };

    for (const [option, count] of distribution) {
      if (count > winner.votes) {
        winner = { option, votes: count };
      }
    }

    return {
      winner,
      voteDistribution: Object.fromEntries(distribution)
    };
  }
}
```
