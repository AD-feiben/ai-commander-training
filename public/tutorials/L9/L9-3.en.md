## 1. OpenAI API Integration

### 1.1 Service Wrapper

```typescript
// services/aiService.ts
import OpenAI from 'openai';

const openai = new OpenAI({
  apiKey: process.env.REACT_APP_OPENAI_API_KEY,
  dangerouslyAllowBrowser: true
});

interface TaskParseResult {
  title: string;
  description?: string;
  priority: 'low' | 'medium' | 'high';
  category: string;
  dueDate?: string;
}

export const aiService = {
  // Parse natural language tasks
  async parseTask(input: string): Promise<TaskParseResult> {
    const prompt = `
Parse the following natural language description into structured task information:

Input: "${input}"

Please extract the following information and return in JSON format:
{
  "title": "Task title (concise and clear)",
  "description": "Task detailed description (optional)",
  "priority": "Priority (low/medium/high, judged by urgency)",
  "category": "Category (work/personal/study/other)",
  "dueDate": "Due date (ISO format, e.g. 2024-01-15, optional)"
}

Notes:
- If no explicit time is given in the input, dueDate is null
- Priority is judged by keywords: urgent/immediately/important → high, tomorrow/this week → medium, others → low
`;

    const response = await openai.chat.completions.create({
      model: 'gpt-3.5-turbo',
      messages: [
        {
          role: 'system',
          content: 'You are a task parsing assistant, skilled at extracting structured information from natural language.'
        },
        {
          role: 'user',
          content: prompt
        }
      ],
      temperature: 0.3,
      response_format: { type: 'json_object' }
    });

    const content = response.choices[0].message.content;
    if (!content) {
      throw new Error('AI response is empty');
    }

    return JSON.parse(content);
  },

  // Auto-categorize
  async classifyTask(title: string, description?: string): Promise<string> {
    const prompt = `
Categorize the following task into the most appropriate category:

Title: ${title}
Description: ${description || 'None'}

Available categories:
- Development (programming, code-related)
- Design (UI/UX, graphic design)
- Meeting (discussion, sync)
- Documentation (writing, document organization)
- Learning (reading, training)
- Personal (life, health)

Only return the category name, nothing else.
`;

    const response = await openai.chat.completions.create({
      model: 'gpt-3.5-turbo',
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.3,
      max_tokens: 20
    });

    return response.choices[0].message.content?.trim() || 'Other';
  }
};
```

### 1.2 Error Handling

```typescript
// services/aiService.ts (continued)

// AI call with retry
async function callAIWithRetry<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  let lastError: Error;

  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error as Error;

      // Exponential backoff
      const delay = Math.pow(2, i) * 1000;
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }

  throw lastError!;
}

// Fallback strategy
export const aiServiceWithFallback = {
  async parseTask(input: string): Promise<TaskParseResult> {
    try {
      return await callAIWithRetry(() => aiService.parseTask(input));
    } catch (error) {
      console.warn('AI parsing failed, using local parsing:', error);

      // Local simple parsing as fallback
      return {
        title: input.slice(0, 50),
        description: input,
        priority: 'medium',
        category: 'Other'
      };
    }
  }
};
```

## 2. Natural Language Task Creation

### 2.1 Smart Input Component

```typescript
// components/tasks/SmartTaskInput.tsx
import { useState } from 'react';
import { aiServiceWithFallback } from '../../services/aiService';
import { useTaskStore } from '../../stores/taskStore';

export function SmartTaskInput() {
  const [input, setInput] = useState('');
  const [isParsing, setIsParsing] = useState(false);
  const [parsedResult, setParsedResult] = useState<any>(null);

  const addTask = useTaskStore(state => state.addTask);

  const handleParse = async () => {
    if (!input.trim()) return;

    setIsParsing(true);
    try {
      const result = await aiServiceWithFallback.parseTask(input);
      setParsedResult(result);
    } catch (error) {
      console.error('Parsing failed:', error);
    } finally {
      setIsParsing(false);
    }
  };

  const handleConfirm = () => {
    if (parsedResult) {
      addTask({
        title: parsedResult.title,
        description: parsedResult.description,
        priority: parsedResult.priority,
        status: 'todo'
      });
      setInput('');
      setParsedResult(null);
    }
  };

  return (
    <div className="space-y-4">
      <div className="relative">
        <textarea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="Describe your task, e.g.: Complete the project report by 3pm tomorrow, very important"
          className="w-full p-3 border rounded-lg resize-none"
          rows={3}
        />
        <button
          onClick={handleParse}
          disabled={isParsing || !input.trim()}
          className="absolute bottom-2 right-2 px-4 py-1 bg-blue-600 text-white rounded text-sm disabled:opacity-50"
        >
          {isParsing ? 'Parsing...' : 'Smart Parse'}
        </button>
      </div>

      {parsedResult && (
        <div className="p-4 bg-gray-50 rounded-lg">
          <h4 className="font-medium mb-2">Parsing Result:</h4>
          <div className="space-y-1 text-sm">
            <p><span className="text-gray-500">Title:</span> {parsedResult.title}</p>
            <p><span className="text-gray-500">Priority:</span> {parsedResult.priority}</p>
            <p><span className="text-gray-500">Category:</span> {parsedResult.category}</p>
            {parsedResult.dueDate && (
              <p><span className="text-gray-500">Due:</span> {parsedResult.dueDate}</p>
            )}
          </div>
          <div className="mt-3 flex gap-2">
            <button
              onClick={handleConfirm}
              className="px-4 py-1 bg-green-600 text-white rounded text-sm"
            >
              Confirm & Create
            </button>
            <button
              onClick={() => setParsedResult(null)}
              className="px-4 py-1 bg-gray-300 text-gray-700 rounded text-sm"
            >
              Re-enter
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```

## 3. Auto-categorization and Priority

### 3.1 Batch Processing

```typescript
// services/autoOrganizeService.ts
import { aiService } from './aiService';
import { useTaskStore } from '../stores/taskStore';

export const autoOrganizeService = {
  // Batch categorize uncategorized tasks
  async classifyUncategorizedTasks(): Promise<void> {
    const tasks = useTaskStore.getState().tasks;
    const uncategorized = tasks.filter(t => t.category === 'Other');

    for (const task of uncategorized) {
      try {
        const category = await aiService.classifyTask(
          task.title,
          task.description
        );

        useTaskStore.getState().updateTask(task.id, { category });
      } catch (error) {
        console.error(`Failed to categorize task ${task.id}:`, error);
      }
    }
  },

  // Smart priority suggestion
  async suggestPriority(taskId: string): Promise<void> {
    const task = useTaskStore.getState().tasks.find(t => t.id === taskId);
    if (!task) return;

    const prompt = `
Analyze the following task and suggest priority (low/medium/high):

Title: ${task.title}
Description: ${task.description || 'None'}

Considerations:
- Keywords: urgent, important, immediately, now → high
- Keywords: tomorrow, this week, planned → medium
- Others → low

Only return the priority word.
`;

    // Call AI for suggestions
    // Can display to user for confirmation
  }
};
```
