## 1. Team Management

### 1.1 Permission Model

**Permission Hierarchy**:

- **Organization**
  - **Team**
    - **Admin** — Full permissions
    - **Member**
      - Create tasks
      - Edit own tasks
      - Comment
    - **Guest** — Read-only

### 1.2 Team Store

```typescript
// stores/teamStore.ts
import { create } from 'zustand';

interface Team {
  id: string;
  name: string;
  members: TeamMember[];
  inviteCode?: string;
}

interface TeamMember {
  userId: string;
  name: string;
  email: string;
  avatar?: string;
  role: 'admin' | 'member' | 'guest';
  joinedAt: Date;
}

interface TeamState {
  currentTeam: Team | null;
  teams: Team[];

  createTeam: (name: string) => Promise<void>;
  joinTeam: (inviteCode: string) => Promise<void>;
  inviteMember: (email: string, role: TeamMember['role']) => Promise<void>;
  removeMember: (userId: string) => Promise<void>;
  updateMemberRole: (userId: string, role: TeamMember['role']) => Promise<void>;

  // Permission checks
  canManageTeam: () => boolean;
  canAssignTasks: () => boolean;
}

export const useTeamStore = create<TeamState>((set, get) => ({
  currentTeam: null,
  teams: [],

  createTeam: async (name) => {
    // API call to create team
    const newTeam: Team = {
      id: crypto.randomUUID(),
      name,
      members: []
    };
    set({ currentTeam: newTeam });
  },

  joinTeam: async (inviteCode) => {
    // Join team via invite code
  },

  inviteMember: async (email, role) => {
    const { currentTeam } = get();
    if (!currentTeam || !get().canManageTeam()) {
      throw new Error('No permission');
    }

    // Send invitation
    // API call
  },

  removeMember: async (userId) => {
    const { currentTeam } = get();
    if (!currentTeam || !get().canManageTeam()) return;

    set({
      currentTeam: {
        ...currentTeam,
        members: currentTeam.members.filter(m => m.userId !== userId)
      }
    });
  },

  updateMemberRole: async (userId, role) => {
    const { currentTeam } = get();
    if (!currentTeam || !get().canManageTeam()) return;

    set({
      currentTeam: {
        ...currentTeam,
        members: currentTeam.members.map(m =>
          m.userId === userId ? { ...m, role } : m
        )
      }
    });
  },

  canManageTeam: () => {
    const { currentTeam } = get();
    if (!currentTeam) return false;

    const currentUserId = useAuthStore.getState().user?.id;
    const member = currentTeam.members.find(m => m.userId === currentUserId);
    return member?.role === 'admin';
  },

  canAssignTasks: () => {
    const { currentTeam } = get();
    if (!currentTeam) return true; // Personal mode

    const currentUserId = useAuthStore.getState().user?.id;
    const member = currentTeam.members.find(m => m.userId === currentUserId);
    return member?.role === 'admin' || member?.role === 'member';
  }
}));
```

## 2. Task Assignment

### 2.1 Task Assignee Component

```typescript
// components/tasks/TaskAssignee.tsx
import { useTeamStore } from '../../stores/teamStore';
import { useTaskStore } from '../../stores/taskStore';

interface TaskAssigneeProps {
  taskId: string;
  currentAssigneeId?: string;
}

export function TaskAssignee({ taskId, currentAssigneeId }: TaskAssigneeProps) {
  const { currentTeam, canAssignTasks } = useTeamStore();
  const updateTask = useTaskStore(state => state.updateTask);

  if (!currentTeam) return null;
  if (!canAssignTasks()) {
    // Read-only display
    const assignee = currentTeam.members.find(m => m.userId === currentAssigneeId);
    return (
      <div className="flex items-center gap-2">
        {assignee ? (
          <>
            <img src={assignee.avatar} className="w-6 h-6 rounded-full" />
            <span>{assignee.name}</span>
          </>
        ) : (
          <span className="text-gray-400">Unassigned</span>
        )}
      </div>
    );
  }

  return (
    <select
      value={currentAssigneeId || ''}
      onChange={(e) => updateTask(taskId, { assigneeId: e.target.value || undefined })}
      className="border rounded px-2 py-1"
    >
      <option value="">Unassigned</option>
      {currentTeam.members.map(member => (
        <option key={member.userId} value={member.userId}>
          {member.name}
        </option>
      ))}
    </select>
  );
}
```

## 3. Comment System

### 3.1 Comment Store

```typescript
// stores/commentStore.ts
import { create } from 'zustand';

interface Comment {
  id: string;
  taskId: string;
  userId: string;
  userName: string;
  userAvatar?: string;
  content: string;
  createdAt: Date;
  mentions?: string[]; // Mentioned user IDs
}

interface CommentState {
  comments: Comment[];

  addComment: (taskId: string, content: string) => Promise<void>;
  deleteComment: (commentId: string) => Promise<void>;
  getTaskComments: (taskId: string) => Comment[];
}

export const useCommentStore = create<CommentState>((set, get) => ({
  comments: [],

  addComment: async (taskId, content) => {
    const user = useAuthStore.getState().user;
    if (!user) throw new Error('Not logged in');

    // Parse @mentions
    const mentions = content.match(/@(\w+)/g)?.map(m => m.slice(1)) || [];

    const newComment: Comment = {
      id: crypto.randomUUID(),
      taskId,
      userId: user.id,
      userName: user.name,
      userAvatar: user.avatar,
      content,
      createdAt: new Date(),
      mentions
    };

    set(state => ({
      comments: [...state.comments, newComment]
    }));

    // Send notifications to mentioned people
    if (mentions.length > 0) {
      await notificationService.sendMentionNotifications(mentions, newComment);
    }
  },

  deleteComment: async (commentId) => {
    set(state => ({
      comments: state.comments.filter(c => c.id !== commentId)
    }));
  },

  getTaskComments: (taskId) => {
    return get().comments
      .filter(c => c.taskId === taskId)
      .sort((a, b) => b.createdAt.getTime() - a.createdAt.getTime());
  }
}));
```

### 3.2 Comment Component

```typescript
// components/tasks/TaskComments.tsx
import { useState } from 'react';
import { useCommentStore } from '../../stores/commentStore';
import { useTeamStore } from '../../stores/teamStore';

interface TaskCommentsProps {
  taskId: string;
}

export function TaskComments({ taskId }: TaskCommentsProps) {
  const [newComment, setNewComment] = useState('');
  const [showMentions, setShowMentions] = useState(false);

  const comments = useCommentStore(state => state.getTaskComments(taskId));
  const addComment = useCommentStore(state => state.addComment);
  const { currentTeam } = useTeamStore();

  const handleSubmit = async () => {
    if (!newComment.trim()) return;
    await addComment(taskId, newComment);
    setNewComment('');
  };

  const insertMention = (userName: string) => {
    setNewComment(prev => `${prev}@${userName} `);
    setShowMentions(false);
  };

  return (
    <div className="space-y-4">
      <h3 className="font-medium">Comments ({comments.length})</h3>

      {/* Comment List */}
      <div className="space-y-3">
        {comments.map(comment => (
          <div key={comment.id} className="flex gap-3">
            <img
              src={comment.userAvatar || '/default-avatar.png'}
              className="w-8 h-8 rounded-full"
            />
            <div className="flex-1">
              <div className="flex items-center gap-2">
                <span className="font-medium">{comment.userName}</span>
                <span className="text-xs text-gray-400">
                  {formatDate(comment.createdAt)}
                </span>
              </div>
              <p className="text-sm mt-1">{comment.content}</p>
            </div>
          </div>
        ))}
      </div>

      {/* Add Comment */}
      <div className="relative">
        <textarea
          value={newComment}
          onChange={(e) => setNewComment(e.target.value)}
          placeholder="Add a comment, use @ to mention team members"
          className="w-full p-3 border rounded-lg resize-none"
          rows={3}
        />

        {/* Mention Dropdown */}
        {showMentions && currentTeam && (
          <div className="absolute bottom-full mb-1 bg-white border rounded-lg shadow-lg">
            {currentTeam.members.map(member => (
              <button
                key={member.userId}
                onClick={() => insertMention(member.name)}
                className="flex items-center gap-2 px-4 py-2 hover:bg-gray-100 w-full"
              >
                <img src={member.avatar} className="w-6 h-6 rounded-full" />
                <span>{member.name}</span>
              </button>
            ))}
          </div>
        )}

        <div className="flex justify-between items-center mt-2">
          <button
            onClick={() => setShowMentions(!showMentions)}
            className="text-sm text-blue-600"
          >
            @ Mention
          </button>
          <button
            onClick={handleSubmit}
            disabled={!newComment.trim()}
            className="px-4 py-1 bg-blue-600 text-white rounded text-sm disabled:opacity-50"
          >
            Send
          </button>
        </div>
      </div>
    </div>
  );
}
```
