## 一、OpenAI API 集成

### 1.1 服务封装

```typescript
// services/aiService.ts
import OpenAI from 'openai';

const openai = new OpenAI({
  apiKey: process.env.REACT_APP_OPENAI_API_KEY,
  dangerouslyAllowBrowser: true
});

interface TaskParseResult {
  title: string;
  description?: string;
  priority: 'low' | 'medium' | 'high';
  category: string;
  dueDate?: string;
}

export const aiService = {
  // 解析自然语言任务
  async parseTask(input: string): Promise<TaskParseResult> {
    const prompt = `
将以下自然语言描述解析为结构化任务信息：

输入："${input}"

请提取以下信息并以 JSON 格式返回：
{
  "title": "任务标题（简洁明了）",
  "description": "任务详细描述（可选）",
  "priority": "优先级（low/medium/high，根据紧急程度判断）",
  "category": "分类（工作/个人/学习/其他）",
  "dueDate": "截止日期（ISO 格式，如 2024-01-15，可选）"
}

注意：
- 如果输入中没有明确时间，dueDate 为 null
- 优先级根据词语判断：马上/紧急/重要 → high，明天/本周 → medium，其他 → low
`;

    const response = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [
        {
          role: 'system',
          content: '你是一个任务解析助手，擅长从自然语言中提取结构化信息。'
        },
        {
          role: 'user',
          content: prompt
        }
      ],
      temperature: 0.3,
      response_format: { type: 'json_object' }
    });

    const content = response.choices[0].message.content;
    if (!content) {
      throw new Error('AI 响应为空');
    }

    return JSON.parse(content);
  },

  // 自动分类
  async classifyTask(title: string, description?: string): Promise<string> {
    const prompt = `
将以下任务分类到最合适的类别：

标题：${title}
描述：${description || '无'}

可选类别：
- 开发（编程、代码相关）
- 设计（UI/UX、图形设计）
- 会议（讨论、同步）
- 文档（写作、文档整理）
- 学习（阅读、培训）
- 个人（生活、健康）

只返回类别名称，不要其他内容。
`;

    const response = await openai.chat.completions.create({
      model: 'gpt-4o-mini',
      messages: [{ role: 'user', content: prompt }],
      temperature: 0.3,
      max_tokens: 20
    });

    return response.choices[0].message.content?.trim() || '其他';
  }
};
```

### 1.2 错误处理

```typescript
// services/aiService.ts（续）

// 带重试的 AI 调用
async function callAIWithRetry<T>(
  fn: () => Promise<T>,
  maxRetries = 3
): Promise<T> {
  let lastError: Error;
  
  for (let i = 0; i < maxRetries; i++) {
    try {
      return await fn();
    } catch (error) {
      lastError = error as Error;
      
      // 指数退避
      const delay = Math.pow(2, i) * 1000;
      await new Promise(resolve => setTimeout(resolve, delay));
    }
  }
  
  throw lastError!;
}

// 降级策略
export const aiServiceWithFallback = {
  async parseTask(input: string): Promise<TaskParseResult> {
    try {
      return await callAIWithRetry(() => aiService.parseTask(input));
    } catch (error) {
      console.warn('AI 解析失败，使用本地解析:', error);
      
      // 本地简单解析作为降级
      return {
        title: input.slice(0, 50),
        description: input,
        priority: 'medium',
        category: '其他'
      };
    }
  }
};
```

## 二、自然语言任务创建

### 2.1 智能输入组件

```typescript
// components/tasks/SmartTaskInput.tsx
import { useState } from 'react';
import { aiServiceWithFallback } from '../../services/aiService';
import { useTaskStore } from '../../stores/taskStore';

export function SmartTaskInput() {
  const [input, setInput] = useState('');
  const [isParsing, setIsParsing] = useState(false);
  const [parsedResult, setParsedResult] = useState<any>(null);
  
  const addTask = useTaskStore(state => state.addTask);
  
  const handleParse = async () => {
    if (!input.trim()) return;
    
    setIsParsing(true);
    try {
      const result = await aiServiceWithFallback.parseTask(input);
      setParsedResult(result);
    } catch (error) {
      console.error('解析失败:', error);
    } finally {
      setIsParsing(false);
    }
  };
  
  const handleConfirm = () => {
    if (parsedResult) {
      addTask({
        title: parsedResult.title,
        description: parsedResult.description,
        priority: parsedResult.priority,
        status: 'todo'
      });
      setInput('');
      setParsedResult(null);
    }
  };
  
  return (
    <div className="space-y-4">
      <div className="relative">
        <textarea
          value={input}
          onChange={(e) => setInput(e.target.value)}
          placeholder="描述你的任务，例如：明天下午3点前完成项目报告，很重要"
          className="w-full p-3 border rounded-lg resize-none"
          rows={3}
        />
        <button
          onClick={handleParse}
          disabled={isParsing || !input.trim()}
          className="absolute bottom-2 right-2 px-4 py-1 bg-blue-600 text-white rounded text-sm disabled:opacity-50"
        >
          {isParsing ? '解析中...' : '智能解析'}
        </button>
      </div>
      
      {parsedResult && (
        <div className="p-4 bg-gray-50 rounded-lg">
          <h4 className="font-medium mb-2">解析结果：</h4>
          <div className="space-y-1 text-sm">
            <p><span className="text-gray-500">标题：</span>{parsedResult.title}</p>
            <p><span className="text-gray-500">优先级：</span>{parsedResult.priority}</p>
            <p><span className="text-gray-500">分类：</span>{parsedResult.category}</p>
            {parsedResult.dueDate && (
              <p><span className="text-gray-500">截止：</span>{parsedResult.dueDate}</p>
            )}
          </div>
          <div className="mt-3 flex gap-2">
            <button
              onClick={handleConfirm}
              className="px-4 py-1 bg-green-600 text-white rounded text-sm"
            >
              确认创建
            </button>
            <button
              onClick={() => setParsedResult(null)}
              className="px-4 py-1 bg-gray-300 text-gray-700 rounded text-sm"
            >
              重新输入
            </button>
          </div>
        </div>
      )}
    </div>
  );
}
```

## 三、自动分类和优先级

### 3.1 批量处理

```typescript
// services/autoOrganizeService.ts
import { aiService } from './aiService';
import { useTaskStore } from '../stores/taskStore';

export const autoOrganizeService = {
  // 批量分类未分类任务
  async classifyUncategorizedTasks(): Promise<void> {
    const tasks = useTaskStore.getState().tasks;
    const uncategorized = tasks.filter(t => t.category === '其他');
    
    for (const task of uncategorized) {
      try {
        const category = await aiService.classifyTask(
          task.title,
          task.description
        );
        
        useTaskStore.getState().updateTask(task.id, { category });
      } catch (error) {
        console.error(`分类任务 ${task.id} 失败:`, error);
      }
    }
  },

- 关键词：紧急、重要、马上、立即 → high
- 关键词：明天、本周、计划 → medium
```
