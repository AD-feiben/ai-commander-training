## 3. MCP Protocol Deep Dive

### 3.1 What is MCP

**MCP (Model Context Protocol)** is an open standard protocol introduced by Anthropic that enables AI to securely connect to external tools and data sources.

**Core Design Philosophy:**
- **Standardization**: Unified interface, tools interchangeable
- **Security**: User controls permissions
- **Scalability**: Easy to add new tools

### 3.2 MCP Architecture


### 3.3 MCP Server Example

**File System Server:**

```typescript
// mcp-filesystem-server.ts
import { Server } from '@modelcontextprotocol/sdk/server/index.js';
import { StdioServerTransport } from '@modelcontextprotocol/sdk/server/stdio.js';

const server = new Server({
  name: 'filesystem-server',
  version: '1.0.0'
}, {
  capabilities: {
    tools: {}
  }
});

// Define available tools
server.setRequestHandler('tools/list', async () => {
  return {
    tools: [
      {
        name: 'read_file',
        description: 'Read file content',
        inputSchema: {
          type: 'object',
          properties: {
            path: { type: 'string', description: 'File path' }
          },
          required: ['path']
        }
      },
      {
        name: 'write_file',
        description: 'Write file content',
        inputSchema: {
          type: 'object',
          properties: {
            path: { type: 'string', description: 'File path' },
            content: { type: 'string', description: 'File content' }
          },
          required: ['path', 'content']
        }
      },
      {
        name: 'list_directory',
        description: 'List directory contents',
        inputSchema: {
          type: 'object',
          properties: {
            path: { type: 'string', description: 'Directory path' }
          },
          required: ['path']
        }
      }
    ]
  };
});

// Handle tool calls
server.setRequestHandler('tools/call', async (request) => {
  const { name, arguments: args } = request.params;
  
  switch (name) {
    case 'read_file':
      const content = await fs.readFile(args.path, 'utf-8');
      return {
        content: [{ type: 'text', text: content }]
      };
    
    case 'write_file':
      await fs.writeFile(args.path, args.content);
      return {
        content: [{ type: 'text', text: 'File written successfully' }]
      };
    
    case 'list_directory':
      const entries = await fs.readdir(args.path, { withFileTypes: true });
      const result = entries.map(e => ({
        name: e.name,
        type: e.isDirectory() ? 'directory' : 'file'
      }));
      return {
        content: [{ type: 'text', text: JSON.stringify(result, null, 2) }]
      };
    
    default:
      throw new Error(`Unknown tool: ${name}`);
  }
});

// Start server
const transport = new StdioServerTransport();
await server.connect(transport);
```

### 4.1 Function Calling Principle


```typescript
const openai = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

const functions = [

        },

          type: 'string',

        }
      },

    }
  },
  {
    name: 'run_tests',
    description: 'Run unit tests',
    parameters: {
      type: 'object',
      properties: {
        test_pattern: {
          type: 'string',
          description: 'Test file matching pattern'
        },
        coverage: {
          type: 'boolean',
          description: 'Whether to generate coverage report'
        }
      }
    }
  }
];

// Call AI
async function chatWithFunctions(userMessage: string) {
  const messages = [
    { role: 'user', content: userMessage }
  ];
  
  const response = await openai.chat.completions.create({
    model: 'gpt-4',
    messages,
    functions,
    function_call: 'auto'
  });
  
  const message = response.choices[0].message;
  
  // Check if function call is needed
  if (message.function_call) {
    const functionName = message.function_call.name;
    const args = JSON.parse(message.function_call.arguments);
    
    console.log(`AI decided to call function: ${functionName}`);
    console.log(`Arguments:`, args);
    
    // Execute function
    const functionResult = await executeFunction(functionName, args);
    
    // Return result to AI
    messages.push(message);
    messages.push({
      role: 'function',
      name: functionName,
      content: JSON.stringify(functionResult)
    });
    
    // Get final response
    const finalResponse = await openai.chat.completions.create({
      model: 'gpt-4',
      messages
    });
    
    return finalResponse.choices[0].message.content;
  }
  
  return message.content;
}

// Function execution mapping
async function executeFunction(name: string, args: any) {
  switch (name) {
    case 'get_current_weather':
      return await getWeather(args.location, args.unit);
    case 'search_code':
      return await searchCode(args.query, args.file_type);
    case 'run_tests':
      return await runTests(args.test_pattern, args.coverage);
    default:
      throw new Error(`Unknown function: ${name}`);
  }
}
```

### 4.3 Multi-turn Function Calling

```typescript
// Handle complex multi-step tasks
async function handleComplexTask(userRequest: string) {
  const messages = [
    {
      role: 'system',
      content: 'You are a development assistant, helping users analyze code, run tests, and fix issues.'
    },
    { role: 'user', content: userRequest }
  ];
  
  let maxIterations = 10;
  
  while (maxIterations > 0) {
    const response = await openai.chat.completions.create({
      model: 'gpt-4',
      messages,
      functions,
      function_call: 'auto'
    });
    
    const message = response.choices[0].message;
    
    // If no function call, task is complete
    if (!message.function_call) {
      return message.content;
    }
    
    // Execute function call
    const functionName = message.function_call.name;
    const args = JSON.parse(message.function_call.arguments);
    const result = await executeFunction(functionName, args);
    
    // Add to message history
    messages.push(message);
    messages.push({
      role: 'function',
      name: functionName,
      content: JSON.stringify(result)
    });
    
    maxIterations--;
  }
  
  return 'Maximum iterations reached';
}

// Usage example
const result = await handleComplexTask(`
  I found performance issues in the project. Please help me:
  1. Search all components using useEffect
  2. Analyze if there's room for optimization
  3. Run tests to ensure no issues after changes
`);
```

## 5. Practice: Develop File Operation Tool

### 5.1 Tool Design

**Goal:** Develop a file operation tool supporting code review

**Features:**
- Read project files
- Search code
- Analyze file structure
- Generate file reports

### 5.2 Tool Implementation

```typescript
// tools/FileTool.ts
import * as fs from 'fs/promises';
import * as path from 'path';
import { glob } from 'glob';

export interface FileToolConfig {
  rootPath: string;
  allowedExtensions: string[];
  maxFileSize: number; // bytes
}

export class FileTool {
  private config: FileToolConfig;
  
  constructor(config: FileToolConfig) {
    this.config = config;
  }
  
  // Read file content
  async readFile(filePath: string): Promise<string> {
    const fullPath = path.join(this.config.rootPath, filePath);
    
    // Security check: ensure within root directory
    if (!fullPath.startsWith(this.config.rootPath)) {
      throw new Error('Access denied: path exceeds project scope');
    }
    
    // Check file extension
    const ext = path.extname(filePath);
    if (!this.config.allowedExtensions.includes(ext)) {
      throw new Error(`Unsupported file type: ${ext}`);
    }
    
    // Check file size
    const stats = await fs.stat(fullPath);
    if (stats.size > this.config.maxFileSize) {
      throw new Error(`File too large: ${stats.size} bytes`);
    }
    
    return await fs.readFile(fullPath, 'utf-8');
  }
  
  // Search code
  async searchCode(pattern: string, filePattern?: string): Promise<SearchResult[]> {
    const files = await glob(filePattern || '**/*.{ts,tsx,js,jsx}', {
      cwd: this.config.rootPath,
      ignore: ['node_modules/**', 'dist/**', '.git/**']
    });
    
    const results: SearchResult[] = [];
    
    for (const file of files) {
      const content = await this.readFile(file);
      const lines = content.split('\n');
      
      lines.forEach((line, index) => {
        if (line.includes(pattern)) {
          results.push({
            file,
            line: index + 1,
            content: line.trim()
          });
        }
      });
    }
    
    return results;
  }
  
  // Analyze project structure
  async analyzeProjectStructure(): Promise<ProjectStructure> {
    const structure: ProjectStructure = {
      directories: [],
      files: [],
      stats: {
        totalFiles: 0,
        totalLines: 0,
        byExtension: {}
      }
    };
    
    const files = await glob('**/*', {
      cwd: this.config.rootPath,
      ignore: ['node_modules/**', 'dist/**', '.git/**'],
      nodir: false
    });
    
    for (const file of files) {
      const fullPath = path.join(this.config.rootPath, file);
      const stats = await fs.stat(fullPath);
      
      if (stats.isDirectory()) {
        structure.directories.push(file);
      } else {
        const ext = path.extname(file);
        structure.stats.byExtension[ext] = (structure.stats.byExtension[ext] || 0) + 1;
        
        if (this.config.allowedExtensions.includes(ext)) {
          const content = await this.readFile(file);
          const lines = content.split('\n').length;
          
          structure.files.push({
            path: file,
            size: stats.size,
            lines
          });
          
          structure.stats.totalFiles++;
          structure.stats.totalLines += lines;
        }
      }
    }
    
    return structure;
  }
  
  // Generate file report
  async generateReport(outputPath: string): Promise<void> {
    const structure = await this.analyzeProjectStructure();
    
    const report = `
# Project Code Analysis Report

Generated: ${new Date().toISOString()}

## Statistics Overview

- Total files: ${structure.stats.totalFiles}
- Total lines: ${structure.stats.totalLines}
- Average lines per file: ${Math.round(structure.stats.totalLines / structure.stats.totalFiles)}

## File Type Distribution

${Object.entries(structure.stats.byExtension)
  .map(([ext, count]) => `- ${ext || 'No extension'}: ${count} files`)
  .join('\n')}

## Top 10 Largest Files

${structure.files
  .sort((a, b) => b.lines - a.lines)
  .slice(0, 10)
  .map((f, i) => `${i + 1}. ${f.path} (${f.lines} lines)`)
  .join('\n')}
`;
    
    await fs.writeFile(outputPath, report);
  }
}

// Type definitions
interface SearchResult {
  file: string;
  line: number;
  content: string;
}

interface ProjectStructure {
  directories: string[];
  files: Array<{
    path: string;
    size: number;
    lines: number;
  }>;
  stats: {
    totalFiles: number;
    totalLines: number;
    byExtension: Record<string, number>;
  };
}
```

### 5.3 Integrate into Skill

```typescript
// skills/CodeReviewSkill.ts
import { FileTool } from '../tools/FileTool';
import { OpenAI } from 'openai';

export class CodeReviewSkill {
  private fileTool: FileTool;
  private openai: OpenAI;
  
  constructor(config: SkillConfig) {
    this.fileTool = new FileTool(config.fileTool);
    this.openai = new OpenAI({ apiKey: config.openaiApiKey });
  }
  
  async reviewFile(filePath: string): Promise<ReviewResult> {
    // 1. Read file
    const code = await this.fileTool.readFile(filePath);
    
    // 2. Analyze code
    const analysis = await this.analyzeCode(code, filePath);
    
    // 3. Generate report
    const report = await this.generateReport(analysis);
    
    return {
      file: filePath,
      issues: analysis.issues,
      suggestions: analysis.suggestions,
      report
    };
  }
  
  async reviewProject(): Promise<ProjectReviewResult> {
    // 1. Analyze project structure
    const structure = await this.fileTool.analyzeProjectStructure();
    
    // 2. Identify key files
    const keyFiles = this.identifyKeyFiles(structure);
    
    // 3. Review key files
    const results: ReviewResult[] = [];
    for (const file of keyFiles) {
      const result = await this.reviewFile(file);
      results.push(result);
    }
    
    // 4. Generate project-level report
    const projectReport = await this.generateProjectReport(results, structure);
    
    return {
      structure,
      fileReviews: results,
      projectReport
    };
  }
  
  private async analyzeCode(code: string, filePath: string): Promise<CodeAnalysis> {
    const response = await this.openai.chat.completions.create({
      model: 'gpt-4',
      messages: [
        {
          role: 'system',
          content: 'You are a code review expert. Please analyze the code and provide improvement suggestions.'
        },
        {
          role: 'user',
          content: `
Please analyze the following code file: ${filePath}

\`\`\`
${code}
\`\`\`

Please analyze from the following dimensions:
1. Security issues
2. Performance issues
3. Code standards
4. Maintainability

Return results in JSON format.
          `
        }
      ],
      response_format: { type: 'json_object' }
    });
    
    return JSON.parse(response.choices[0].message.content);
  }
  
  private identifyKeyFiles(structure: ProjectStructure): string[] {
    // Identify key files: entry files, core modules, large files, etc.
    return structure.files
      .filter(f => {
        // Entry files
        if (f.path.includes('index.') || f.path.includes('main.')) return true;
        // Large files (over 300 lines)
        if (f.lines > 300) return true;
        // Core modules
        if (f.path.includes('/core/') || f.path.includes('/utils/')) return true;
        return false;
      })
      .map(f => f.path)
      .slice(0, 20); // Review max 20 files
  }
}
```

## 6. Tool Security & Permission Control

### 6.1 Security Principles

```
┌─────────────────────────────────────────────────────────────┐
│                    Tool Security Principles                   │
├─────────────────────────────────────────────────────────────┤
│                                                             │
│  1. Principle of Least Privilege                            │
│     └── Only grant minimum permissions needed for task       │
│                                                             │
│  2. Path Restrictions                                        │
│     └── Restrict accessible file system scope               │
│                                                             │
│  3. Operation Auditing                                       │
│     └── Log all tool calls for traceability                 │
│                                                             │
│  4. User Confirmation                                       │
│     └── Sensitive operations require user confirmation      │
│                                                             │
│  5. Sandboxed Execution                                     │
│     └── Execute untrusted code in isolated environment      │
│                                                             │
└─────────────────────────────────────────────────────────────┘
```

### 6.2 Permission Configuration Example

```yaml
# permissions.yaml
tools:
  file_system:
    allowed_paths:
      - ./src
      - ./tests
      - ./docs
    denied_paths:
      - ./node_modules
      - ./.env
      - ./secrets
    allowed_operations:
      - read
      - write
    denied_operations:
      - delete
      - execute
  
  command_execution:
    allowed_commands:
      - npm test
      - npm run build
      - git status
    denied_commands:
      - rm -rf
      - sudo
    require_confirmation: true
  
  api_calls:
    allowed_hosts:
      - api.github.com
      - api.openai.com
    rate_limit: 100/hour
```
