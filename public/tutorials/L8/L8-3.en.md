## 1. Frontend Performance Optimization

### 1.1 Load Optimization

```javascript
// 1. Code splitting
const Dashboard = lazy(() => import('./Dashboard'));

// 2. Preload critical resources
<link rel="preload" href="/critical.css" as="style">
<link rel="prefetch" href="/next-page.js">

// 3. Image optimization
<img 
  src="image.webp"
  loading="lazy"
  decoding="async"
  width="800"
  height="600"
/>
```

### 1.2 Render Optimization

```javascript
// 1. Use useMemo for computed value caching
const processedData = useMemo(() => {
  return data.map(processItem);
}, [data]);

// 2. Use useCallback for function caching
const handleClick = useCallback(() => {
  onItemClick(id);
}, [id, onItemClick]);

// 3. Virtual list
import { FixedSizeList } from 'react-window';

<FixedSizeList
  height={500}
  itemCount={10000}
  itemSize={50}
>
  {Row}
</FixedSizeList>

// 4. Debouncing and throttling
const debouncedSearch = useMemo(
  () => debounce(handleSearch, 300),
  [handleSearch]
);
```

### 1.3 Network Optimization

```javascript
// 1. Use Service Worker for caching
// sw.js
self.addEventListener('fetch', (event) => {
  event.respondWith(
    caches.match(event.request).then((response) => {
      return response || fetch(event.request);
    })
  );
});

// 2. HTTP caching strategy
// Cache-Control: public, max-age=31536000, immutable
```

## 2. Backend Performance Optimization

### 2.1 Database Optimization

```javascript
// 1. Index optimization
// Create composite index
db.collection('users').createIndex({ email: 1, status: 1 });

// 2. Query optimization
// ❌ Avoid
const users = await User.find();
const activeUsers = users.filter(u => u.status === 'active');

// ✅ Recommended
const activeUsers = await User.find({ status: 'active' });

// 3. Pagination optimization
// Use cursor-based pagination
db.collection('posts')
  .find({ _id: { $lt: lastId } })
  .limit(20)
  .sort({ _id: -1 });
```

### 2.2 Caching Strategy

```javascript
// 1. Redis caching
const cacheKey = `user:${userId}`;
let user = await redis.get(cacheKey);

if (!user) {
  user = await db.users.findById(userId);
  await redis.setex(cacheKey, 3600, JSON.stringify(user));
}

// 2. Local caching
const localCache = new Map();
const CACHE_TTL = 5 * 60 * 1000; // 5 minutes

function getCached(key, fetcher) {
  const cached = localCache.get(key);
  if (cached && Date.now() - cached.time < CACHE_TTL) {
    return cached.data;
  }
  
  const data = fetcher();
  localCache.set(key, { data, time: Date.now() });
  return data;
}
```

### 2.3 Asynchronous Processing

```javascript
// 1. Message queue
const queue = new Bull('email-queue');

// Producer
queue.add('send-email', { to, subject, body });

// Consumer
queue.process('send-email', async (job) => {
  await sendEmail(job.data);
});

// 2. Batch processing
const batchProcessor = new BatchProcessor({
  batchSize: 100,
  flushInterval: 1000,
  processor: async (items) => {
    await db.bulkInsert(items);
  }
});
```

## 3. Database Optimization Strategies

### 3.1 Query Optimization

```sql
-- 1. Use EXPLAIN to analyze queries
EXPLAIN ANALYZE SELECT * FROM users WHERE email = 'test@example.com';

-- 2. Avoid SELECT *
SELECT id, name, email FROM users WHERE status = 'active';

-- 3. Use JOIN instead of subqueries
-- ❌ Avoid
SELECT * FROM orders WHERE user_id IN (SELECT id FROM users WHERE status = 'active');

-- ✅ Recommended
SELECT o.* FROM orders o
JOIN users u ON o.user_id = u.id
WHERE u.status = 'active';
```

### 3.2 Index Strategy

```sql
-- 1. Single-column index
CREATE INDEX idx_users_email ON users(email);

-- 2. Composite index
CREATE INDEX idx_orders_user_date ON orders(user_id, created_at);

-- 3. Partial index
CREATE INDEX idx_active_users ON users(email) WHERE status = 'active';

-- 4. Covering index
CREATE INDEX idx_users_covering ON users(email, name, phone);
-- Query SELECT email, name, phone FROM users WHERE email = '...' can get data directly from index
```
