## 1. Monitoring Metrics System

### 1.1 RUM (Real User Monitoring)

```javascript
// Using web-vitals library
import { getCLS, getFID, getFCP, getLCP, getTTFB } from 'web-vitals';

function sendToAnalytics(metric) {
  const body = JSON.stringify(metric);
  
  // Send to analytics service
  fetch('/analytics', {
    body,
    method: 'POST',
    keepalive: true,
  });
}

getCLS(sendToAnalytics);
getFID(sendToAnalytics);
getFCP(sendToAnalytics);
getLCP(sendToAnalytics);
getTTFB(sendToAnalytics);
```

## 2. Monitoring Tools

### 2.1 Frontend Monitoring

```javascript
// Error monitoring
window.addEventListener('error', (event) => {
  reportError({
    type: 'javascript',
    message: event.message,
    filename: event.filename,
    lineno: event.lineno,
    colno: event.colno,
    stack: event.error?.stack
  });
});

// Performance monitoring
const observer = new PerformanceObserver((list) => {
  for (const entry of list.getEntries()) {
    if (entry.entryType === 'largest-contentful-paint') {
      console.log('LCP:', entry.startTime);
    }
  }
});

observer.observe({ entryTypes: ['largest-contentful-paint'] });
```

### 2.2 Backend Monitoring

```javascript
// Express middleware
app.use((req, res, next) => {
  const start = Date.now();
  
  res.on('finish', () => {
    const duration = Date.now() - start;
    
    metrics.timing('http.request', duration, {
      method: req.method,
      route: req.route?.path,
      status: res.statusCode
    });
  });
  
  next();
});

// Database query monitoring
const originalQuery = db.query.bind(db);
db.query = async (...args) => {
  const start = Date.now();
  const result = await originalQuery(...args);
  const duration = Date.now() - start;
  
  metrics.timing('db.query', duration);
  
  return result;
};
```

## 3. Alerting Strategy

### 3.1 Alerting Rules

```yaml
# alerting-rules.yaml
alerts:
  - name: high_error_rate
    condition: error_rate > 5%
    duration: 5m
    severity: critical
    
  - name: slow_response
    condition: p95_response_time > 1000ms
    duration: 10m
    severity: warning
    
  - name: low_apdex
    condition: apdex_score < 0.7
    duration: 15m
    severity: warning
    
  - name: high_cpu
    condition: cpu_usage > 80%
    duration: 10m
    severity: warning
```

### 3.2 Alert Notification

```javascript
// Alert handler
class AlertManager {
  async handleAlert(alert) {
    // Select notification channel based on severity
    if (alert.severity === 'critical') {
      await this.sendPagerDuty(alert);
      await this.sendSlack(alert);
      await this.sendEmail(alert);
    } else if (alert.severity === 'warning') {
      await this.sendSlack(alert);
    }
    
    // Log alert
    await this.logAlert(alert);
  }
  
  // Alert suppression
  shouldSuppress(alert) {
    // Don't send duplicate alerts within 30 minutes
    const key = `${alert.name}:${alert.labels}`;
    const lastAlert = this.alertCache.get(key);
    
    if (lastAlert && Date.now() - lastAlert.time < 30 * 60 * 1000) {
      return true;
    }
    
    this.alertCache.set(key, { time: Date.now() });
    return false;
  }
}
```

- Error rate trend chart
- Throughput trend chart

### 4.2 Implementation Code

```typescript
// Dashboard component
export function PerformanceDashboard() {
  const { data, loading } = usePerformanceData();
  
  if (loading) return <Loading />;
  
  return (
    <div className="dashboard">
      <MetricCards metrics={data.coreWebVitals} />
      <ResponseTimeChart data={data.responseTime} />
      <div className="grid grid-cols-2">
        <ErrorRateChart data={data.errorRate} />
        <ThroughputChart data={data.throughput} />
      </div>
      <AlertList alerts={data.alerts} />
    </div>
  );
}

// Metric cards
function MetricCards({ metrics }) {
  return (
    <div className="metric-cards">
      {metrics.map(metric => (
        <Card key={metric.name}>
          <h3>{metric.name}</h3>
          <div className={metric.status}>
            {metric.value}
          </div>
          <div className="target">
            Target: {metric.target}
          </div>
        </Card>
      ))}
    </div>
  );
}
```
